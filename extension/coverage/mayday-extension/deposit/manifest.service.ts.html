import { Injectable, inject } from '@angular/core';
import { HashedFile, HashingService } from './hashing.service';
 
export interface ManifestInput {
    title: string;
    version: string;
    license: string;
    authorGivenNames: string;
    authorFamilyNames: string;
    authorEmail: string;
    files: HashedFile[];
}
 
export interface BuiltManifest {
    /** YAML text in CITATION.cff 1.2.0 format. */
    yaml: string;
    /** SHA-256 of the YAML — this is what gets timestamped on Bitcoin. */
    sha256: string;
    /** ISO timestamp at which the manifest was issued (local clock). */
    issuedAt: string;
}
 
/**
 * Builds a CITATION.cff manifest from user input + hashed files,
 * then computes the master SHA-256 of the manifest itself.
 *
 * The master hash is what we will eventually submit to OpenTimestamps
 * for Bitcoin anchoring. Until then it serves as the "deposit fingerprint".
 */
@Injectable({ providedIn: 'root' })
export class ManifestService {
    private readonly hashing = inject(HashingService);
 
    async build(input: ManifestInput): Promise<BuiltManifest> {
        const issuedAt = new Date().toISOString();
        const yaml = renderCff(input, issuedAt);
        const sha256 = await this.hashing.hashString(yaml);
        return { yaml, sha256, issuedAt };
    }
}

function yamlString(value: string): string {
    // Simple safe quoting — wrap in double quotes and escape backslash + quote.
    const escaped = (value ?? '').replace(/\\/g, '\\\\').replace(/"/g, '\\"');
    return `"${escaped}"`;
}
